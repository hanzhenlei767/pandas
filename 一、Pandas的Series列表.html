#Series类型
from pandas import Series,DataFrame

obj = Series([1,2,3,4,5])#将list类型转换成Series类型，索引系统默认
print(type(obj))
<class 'pandas.core.series.Series'>
print(obj)#与list相比，Series增加了索引
0    1
1    2
2    3
3    4
4    5

obj2 = Series([6,7,8,9],index = ['d','b','a','c'])#将list类型转换成Series类型，索引index指定
