七、Pandas库（Series/DataFrame直接绘图）

#Pandas中的基本图形

#pandas中的绘图函数
#线图
#Series类型直接绘线图
s = Series(np.random.randn(10).cumsum(), index=np.arange(0, 100, 10))
s.plot()
plt.show()
#DataFrame类型直接绘线图
df = DataFrame(np.random.randn(10, 4).cumsum(0), #10行4列，默认自动画4条直线
               columns=['A', 'B', 'C', 'D'],
               index=np.arange(0, 100, 10))
df.plot()
plt.show()

#柱形图
#Series类型直接绘柱形图
fig, axes = plt.subplots(2, 1)#subplots创建两个图
data = Series(np.random.rand(16), index=list('abcdefghijklmnop'))#index索引是横轴，数据是纵轴
data.plot(kind='bar', ax=axes[0], color='k', alpha=0.7)#bar是垂直的柱形图
data.plot(kind='barh', ax=axes[1], color='k', alpha=0.7)#barh是水平的柱形图
plt.show()
